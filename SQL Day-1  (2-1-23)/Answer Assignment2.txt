- Create Table

Create Table student_dharmik (recid int NOT NULL, Name varchar(255), PRIMARY KEY (recid));

Create Table studentdetail_dharmik (recid int NOT NULL, 
studentid varchar(255) NOT NULL, 
standard int, 
flagsex varchar(255), 
PRIMARY KEY (studentid));

Create Table studentpresent_dharmik (recid int NOT NULL, 
studentid varchar(255) NOT NULL PRIMARY KEY, 
month int, 
present int);

Create Table studenthours_dharmik (
recid int NOT NULL, 
studentid varchar(255) NOT NULL PRIMARY KEY, 
datex date, 
hours int);

 -   Find Average Present of each student.

-   Find Minimum present among all students.
select min(present) as MinPresent from studentpresent_dharmik;

select name, min(present) as MinPresent from student_dharmik, studentpresent_dharmik where student_dharmik.recid is NOT NULL;


-   Find Maximum present among all students.
select max(present) as MaxPresent from studentpresent_dharmik;

select name, max(present) as MaxPresent from student_dharmik, studentpresent_dharmik where student_dharmik.recid is NOT NULL;
