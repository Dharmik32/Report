- Create Table

Create Table student_dharmik (recid int NOT NULL, Name varchar(255), PRIMARY KEY (recid));

Create Table studentdetail_dharmik (recid int NOT NULL, 
studentid varchar(255) NOT NULL, 
standard int, 
flagsex varchar(255), 
PRIMARY KEY (studentid));

Create Table studentpresent_dharmik (recid int NOT NULL, 
studentid varchar(255) NOT NULL PRIMARY KEY, 
month int, 
present int);

Create Table studenthours_dharmik (
recid int NOT NULL, 
studentid varchar(255) NOT NULL PRIMARY KEY, 
datex date, 
hours int);


Create Table product_dharmik (recid int NOT NULL, 
name varchar(255), 
PRIMARY KEY (recid));

Create Table productprice_dharmik (recid int NOT NULL, 
productid int NOT NULL, 
code varchar(255), 
effdt date, 
price int);
