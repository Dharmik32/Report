CREATE TABLE Books_d
(
id INT PRIMARY KEY NOT NULL,
name VARCHAR(50) NOT NULL,
category VARCHAR(50) NOT NULL,
price INT NOT NULL
);

==> create view abc as select name, category from Books_d where category = 'cat3';

category, price from Books_d where category = 'cat5';
